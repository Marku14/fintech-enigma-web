export const LineGraf = () => {
    const [ dataset, setDataset ] = useState([]);
    const [ labels, setLabels ] = useState([]);
    useEffect( () => {
        axios.get( ('https://fondet-fintechenigma.onrender.com/getFunds') )
        .then(res => {
            console.log(res.data);
            if(res.data.status === "OK"){
                setDataset(res.data.dataset.map(e => e.price));
                setLabels(res.data.dataset.map(e => e.time));
                console.log(labels);
            }
            else{
                alert(res.data.status);
            }
        });
    }, []);
    return (
        <div>
            <Line 
                data={{
                    labels: labels,
                    datasets: [{
                        label: 'Fintech Enigma Fondet',
                        data: dataset,
                        borderWidth: 1,
                        fill: false,
                        lineTension: 0.1,
                        borderColor: 'rgb(75, 192, 192)',
                    }]
                }}
                height={400}
                width={600}
                options={{
                    scaleShowLabels : false,
                    maintainAspectRatio: false,
                    scales: {
                        xAxes: [
                        {
                            ticks: {
                                display: true,
                                autoSkip: true,
                                maxTicksLimit: 10,
                                callback: val => {
                                    const [ day, date, Month ] = val.split(" ");
                                    return `${day.slice(0,3)} ${date} ${Month.slice(0,3)}`;
                                    // return `${date} ${Month.slice(0,3)}`;
                                }
                            }
                        }],
                        y: {
                            beginAtZero: false
                        }
                    },
                    elements: {
                        point:{
                            radius: 0
                        }
                    },
                    hover: {
                        intersect: false,
                        mode: 'index'
                    },
                    plugins: {
                        tooltip: {
                          intersect: false
                        }
                      }
                }}
            />
        </div>
    )
}
