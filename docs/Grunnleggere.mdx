# Grunnleggere

import grunnleggere from './folk/grunnleggere.json';
import styles from './css/HovedStyret.module.css';

export const Header = ({children}) => {
    return (
        <h1 className={styles.header}>{children}</h1>
    )
}

export const ClickLeader = link => window.open(link, '_blank')

export const FolkBilder = ({children, name, title, img, link}) => {
    return (
        <div className={styles.leder} onClick={() => ClickLeader(link)}>
            <img alt={link} src={img} className={styles.circle}></img>
            <h3>{name}</h3>
            <h4>{title}</h4>
        </div>
    )
}


export const FolkDiv = ({children, folk}) => {
    return (
        <div className={styles.lederDiv} >
            { folk.map( (props, idx) => (<FolkBilder key={idx} {...props}/>) ) }
        </div>
    )
}
